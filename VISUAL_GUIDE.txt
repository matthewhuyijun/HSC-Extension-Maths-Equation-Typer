🔗 IFRAME PATHS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📄 index.html uses:
   src="../HSC-Extension-Maths-Equation-Typer-main 2/LaTeXLive/index.html"
   
   Explanation:
   ../ = go up one level (from HSC-Extension-Maths-Equation-Typer-main to Desktop)
   HSC-Extension-Maths-Equation-Typer-main 2/LaTeXLive/index.html = navigate to LaTeXLive

📄 working final.html uses:
   src="../../HSC-Extension-Maths-Equation-Typer-main 2/LaTeXLive/html/index.html"
   
   Explanation:
   ../../ = go up TWO levels (from untitled folder → parent → Desktop)
   HSC-Extension-Maths-Equation-Typer-main 2/LaTeXLive/html/index.html = navigate to LaTeXLive
